import React, { useState } from 'react';
import OpenAI from 'openai';

const openai = new OpenAI({apiKey: 'YOUR_API_KEY', dangerouslyAllowBrowser: true}); // Replace with your API key

function FAQCopilot() {
  const [userInput, setUserInput] = useState('');
  const [copilotResponse, setCopilotResponse] = useState('');
  const faqs = {
      // your FAQs here
  };


  const handleInputChange = (event) => {
    setUserInput(event.target.value);
  };

  const handleSubmit = async () => {
    try {
      const completion = await openai.completions.create({
        model: "gpt-3.5-turbo-instruct",
        prompt: `Answer user questions based on the following FAQs:\n${JSON.stringify(faqs)}\n\nUser question: ${userInput}`,
        max_tokens: 150,
      });

      setCopilotResponse(completion.choices[0].text);
    } catch (error) {
      console.error("Error getting response from OpenAI:", error);
      setCopilotResponse("Sorry, I encountered an error. Please try again.");
    }
  };

  return (
    <div>
      <input type="text" value={userInput} onChange={handleInputChange} />
      <button onClick={handleSubmit}>Ask</button>
      <div>{copilotResponse}</div>
    </div>
  );
}

export default FAQCopilot;
